```
# Program to calculate the first 10,000 terms of the series and multiply the result by 4

total_terms = 10000
result = 0.0
sign = 1

for i in range(total_terms):
    term_value = sign / (2 * i + 1)
    result += term_value
    sign *= -1

# Multiply the result by 4
result *= 4

# Print the final result
print(result)

# Output: 3.1414926535900345
```
